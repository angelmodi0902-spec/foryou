<script>
let password = "";
const correct = "190125";
const siteLock = "1122"; // 🔐 CHANGE THIS (site password)

let dots = document.getElementById("dots");

function renderDots() {
    dots.innerHTML = "";
    for (let i = 0; i < 6; i++) {
        let d = document.createElement("div");
        d.className = "dot";
        if (i < password.length) d.classList.add("filled");
        dots.appendChild(d);
    }
}

function addDigit(num) {
    document.getElementById("tap").play();

    if (password.length < 6) {
        password += num;
        renderDots();
    }

    if (password.length === 6) {
        if (password === correct) {
            document.getElementById("success").play();
            setTimeout(() => {
                window.location.href = "success.html";
            }, 500);
        } else {
            password = "";
            renderDots();

            let box = document.getElementById("box");
            box.classList.add("shake");
            setTimeout(() => box.classList.remove("shake"), 400);
        }
    }
}

/* keypad */
let container = document.getElementById("buttons");

for (let i = 1; i <= 9; i++) {
    let btn = document.createElement("button");
    btn.innerText = i;
    btn.onclick = () => addDigit(i);
    container.appendChild(btn);
}

let zero = document.createElement("button");
zero.innerText = 0;
zero.onclick = () => addDigit(0);
container.appendChild(document.createElement("div"));
container.appendChild(zero);

renderDots();

/* 🔐 SITE PASSWORD PROTECTION */
let access = prompt("enter access code 💭");

if (access !== siteLock) {
    document.body.innerHTML = "<h2 style='text-align:center;margin-top:40vh;'>no access 💔</h2>";
}
